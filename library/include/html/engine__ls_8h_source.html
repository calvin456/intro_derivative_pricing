40  //With the technique of estimating continuation values through regression discussed,
41  //one can now present the Longstaff-Schwartz algorithm.
42 
43  double deltaT(TheOption.GetExpiry() / N);
44 
45  // Discount value for each time step:
46 
47  double Disc(exp(-r.Integral(0.0, deltaT)));
48 
49  //1. Start by simulating b independent sample paths Markov chain
50  // Simulate M independent paths first, for regression.
51 
52  Matrix Spots(M, N + 1);
53 
54  MJArray Variates(N + 1);
55  vector<MJArray> args(1);
56  args[0] = Variates;
57 
58  for (size_t i(0); i < M; ++i){
59 
60  MJArray stock_path(N + 1);
61  stock_path = ThePaths->GetOnePath(args);
62 
63  for (size_t j(0); j < N + 1; ++j) Spots[i][j] = stock_path[j];
64 
65  }
66 
67  //2. At the final state, assign each Vmj
68  //Discount all values to become time-0 units
69 
70  Matrix PayOff(M, N);
71 
72  for (size_t i(0); i < M; ++i){
73  for (size_t j(0); j < N; ++j)
74  PayOff[i][j] = std::pow(Disc, static_cast<double>(j + 1)) * TheOption.OptionPayOff(Spots[i][j + 1]);
75  }
76 
77  std::vector<Real> Y(M);
78  Y.assign(PayOff.column_begin(N - 1), PayOff.column_end(N - 1));
79

81  v.push_back(std::bind(weighted_laguerre_poly, std::placeholders::_1, 0));

88

92  // and then Max(payoff, continuation).
93

103 
104  // Estimated continuation values:
105  // Comparing exercise values and continuation values

107 
108  for (size_t j(0); j < M; ++j){
109  double RegVal(0.0);

111

115  }

129 //compute 1st six polynomials
130  switch(a){
131  case 0: {return exp(.5 * x); }
132  case 1: {return exp(.5 * x) * (1.0 - x); }
133  case 2: {return exp(.5 * x) * 1.0 / 2.0 * (std::pow(x,2.0) - 4.0 * x + 2.0); }
134  case 3: {return exp(.5 * x) * 1.0 / 6.0 * (-std::pow(x, 3.0) + 9.0 * std::pow(x, 2.0) - 18.0 * x + 6.0); }
135  case 4: {return exp(.5 * x) * 1.0 / 24.0 * (std::pow(x, 4.0) - 16.0 * std::pow(x, 3.0) + 72.0 * std::pow(x, 2.0) - 96.0 * x + 24.0); }
136  case 5: {return exp(.5 * x) * 1.0 / 120.0 * (-std::pow(x, 5.0) + 25.0 * std::pow(x, 4.0) - 200.0 * std::pow(x, 3.0) + 600.0 * std::pow(x, 2.0) - 600.0 * x + 120.0); }
137  case 6: {return exp(.5 * x) * 1.0 / 720.0 * (std::pow(x, 6.0) - 36.0 * std::pow(x, 5.0) + 450.0 * std::pow(x, 4.0) - 2400.0 * std::pow(x, 3.0) + 5400.0 * std::pow(x, 2.0) - 4320.0 * x + 720.0); }
